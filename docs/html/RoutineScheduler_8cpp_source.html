62 void RoutineScheduler::runRoutine() {
63  // If reached the end, start from the beginning again.
64  if (*mCurrent == nullptr) {
65  mCurrent = Routine::getRoot();
66  // Return if the list is empty
67  if (*mCurrent == nullptr) {
68  return;
69  }
70  }
71 
72 #if ACE_ROUTINE_DEBUG == 1
73  Serial.print(F("Processing "));
74  internal::printRoutineName(&Serial, *mCurrent);
75  Serial.println();
76 #endif
77 
78  // Handle the routine's dispatch back to the last known internal status.
79  switch ((*mCurrent)->getStatus()) {
80  case Routine::kStatusYielding:
81  (*mCurrent)->run();
82  mCurrent = (*mCurrent)->getNext();
83  break;
84  case Routine::kStatusDelaying: {
85  uint16_t elapsedMillis =
86  (*mCurrent)->millis() - (*mCurrent)->getDelayStart();
87  if (elapsedMillis >= (*mCurrent)->getDelay()) {
88  (*mCurrent)->run();
89  }
90  mCurrent = (*mCurrent)->getNext();
91  break;
92  }
93  case Routine::kStatusEnding:
94  // take the routine out of the list, and mark it terminated
95  (*mCurrent)->setTerminated();
96  *mCurrent = *((*mCurrent)->getNext());
97  break;
98  }
99 }

101 void RoutineScheduler::listRoutines(Print* printer) {
102  for (Routine** p = Routine::getRoot(); (*p) != nullptr;
103  p = (*p)->getNext()) {
104  printer->print(F("Routine "));
105  internal::printRoutineName(printer, *p);
106  printer->print(F("; status: "));
